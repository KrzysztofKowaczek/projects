119 template<typename T, int Size>
120 void Matrix<T, Size>::eliminationMethodGauss(Vector<T, Size> *m) const
121 {
122  int row, col;
123  T quotient;
124 
125  for(col = 0; col < Size; col++)
126  for(row = Size - 1; row > col; row--)
127  {
128  if(abs(m[row][col]) > ERR_MARGIN)
129  {
130  quotient = m[row][col] / m[col][col];
131  m[row] -= m[col] * quotient;
132  }
133  }
134 }

139  * wyznacznik macierzy

141 template<typename T, int Size>
142 T Matrix<T, Size>::determinant() const
143 {
144  Vector<T, Size> m[Size]; // Kopia macierzy, na niej wykonywane są wszystkie operacje
145  int rowChanges = 0, place;
146  T det;
147  det = 1;
148  for(int i = 0; i < Size; i++)
149  for(int j = 0; j < Size; j++)
150  m[i][j] = this->_m[j][i]; // Transponowanie macierzy
151 
152  while(checkIfZeroDiagonal(m, place))
153  {
154  int bitmap = 0;
155  for(int i = 0; i < Size; i++)
156  // Oznacz niezerowe komorki w kolumnie
157  if(abs(m[i][place]) > ERR_MARGIN)
158  bitmap |= (1 << i);
159 
160  for(int i = 0; i < Size; i++)
161  if(bitmap & (1 << i)) // Sprawdz ktory jest pierwszy niezerowy
162  {
163  replaceRows(m, place, i); // Zamien wiersz niezerowy z zerowym
164  i = Size; // Zatrzymuje petle
165  }
166  rowChanges++;
167  }
168  eliminationMethodGauss(m);
169  for(int i = 0; i < Size; i++)
170  det *= m[i][i];
171  // Uwzglednienie liczby przestawien wierszy
172  if(rowChanges % 2) det *= -1;
173  return det;
174 }
